00288     public static function preRenderEventsDayAndWeek($mode, $dateStart, $dateEnd, array $eventTypes, array $users, array $userColors) {
00289         $dateStart      = intval($dateStart);
00290         $dateEnd        = intval($dateEnd);
00291         $renderedEvents = array();
00292         $eventFullWidth = $mode === 'day' ? CALENDAR_DAY_EVENT_WIDTH : CALENDAR_WEEK_EVENT_WIDTH;
00293 
00294             // Get all events in current view
00295         $events     = TodoyuEventManager::getEventsInTimespan($dateStart, $dateEnd, $users, $eventTypes, false );
00296             // Group the events by day
00297         $eventsByDay= TodoyuEventManager::groupEventsByDay($events, $dateStart, $dateEnd);
00298             // Add overlap informations to events for each day
00299         $eventsByDay= TodoyuEventManager::addOverlapInformationToEvents($eventsByDay);
00300 
00301             // Render events array
00302         foreach($eventsByDay as $dateKey => $eventsOfDay) {
00303             $dayTime = mktime(0, 0, 0, substr($dateKey, 4, 2), substr($dateKey, 6, 2), substr($dateKey, 0, 4) );
00304 
00305             foreach($eventsOfDay as $event) {
00306 
00307                 if( $event['_overlapNum'] > 0 ) {
00308                         // Event intersects with other events
00309                     $event['width'] = $eventFullWidth / ($event['_overlapNum'] );
00310                     $event['left']  = $event['width'] * $event['_overlapIndex'];
00311                 } else {
00312                         // No intersections to this event
00313                     $event['width'] = $eventFullWidth;
00314                     $event['left']  = 0;
00315                 }
00316 
00317                     // If the event started before the current day, set top = 0
00318                 if( $event['date_start'] <= $dayTime ) {
00319                     $event['top'] = 0;
00320                 } else {
00321                     $event['top']       = TodoyuEventRenderer::getTimeCoordinate($event['date_start']);
00322                 }
00323 
00324                 $event['height']    = TodoyuEventRenderer::getEventHeight($dayTime, $event['date_start'], $event['date_end'] );
00325 
00326                     // Render
00327                 $renderedEvents[$dateKey][] = TodoyuEventRenderer::renderEvent($event, $mode, $users, $userColors);
00328             }

00330

00377     public static function preRenderCalendarBirthdays($calendarMode = 'month', $tstampFirstDay, $tstampLastDay) {
00378         $birthdaysUngrouped   = TodoyuUserManager::getUsersByBirthdayInTimespan($tstampFirstDay, $tstampLastDay);
00379 
00380             // Recompile birthdays into an array of days with their birthdays pre-rendered
00381         $birthdays      = array();
00382         for($tstamp = $tstampFirstDay; $tstamp <= $tstampLastDay; $tstamp += 86400) {
00383             $birthdaysPerDay = '';
00384             $dateMD = date('m-d', $tstamp);
00385             foreach($birthdaysUngrouped as $key => $bdayData) {
00386                 if ($bdayData['birthday_md'] == $dateMD) {
00387                     $bdayData['calendarMode']   = $calendarMode;
00388                     $bdayData['id_user_create'] = -1;
00389                     $bdayData['date_start']     = $tstamp;
00390                     $bdayData['title']          = $bdayData['firstname'] . ' ' . $bdayData['lastname'];
00391 
00392                     $birthdaysPerDay    .= render('ext/calendar/view/birthday.tmpl', $bdayData);
00393                     unset($birthdaysUngrouped[$key]);
00394                 }
00395             }
00396             $birthdays[ date('Ymd', $tstamp) ]  = $birthdaysPerDay;
00397         }
00398 
00399         return $calendarMode == 'day' ? implode('', $birthdays) : $birthdays;
